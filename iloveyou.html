    <script>

        // Lấy tất cả các hình ảnh có trong container

        const images = document.querySelectorAll('.image-container img');

        // Lấy nút bấm

        const showImageBtn = document.getElementById('show-image-btn');

        // Khởi tạo biến đếm để theo dõi ảnh nào sẽ được hiển thị tiếp theo

        let currentImageIndex = 0;


        // Thêm sự kiện 'click' vào nút bấm

        showImageBtn.addEventListener('click', () => {

            // Kiểm tra xem còn ảnh nào chưa hiển thị không

            if (currentImageIndex < images.length) {

                // Hiển thị ảnh tiếp theo bằng cách bỏ class 'hidden'

                images[currentImageIndex].classList.remove('hidden');

                // Tăng biến đếm lên 1

                currentImageIndex++;

            }


            // Nếu tất cả các ảnh đã được hiển thị, ẩn nút bấm đi

            if (currentImageIndex >= images.length) {

                showImageBtn.classList.add('hidden');

            }

        });

    </script>
