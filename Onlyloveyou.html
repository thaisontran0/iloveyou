    <script>

        // Get all images in the container

        const images = document.querySelectorAll('.image-container img');

        // Get the button

        const showImageBtn = document.getElementById('show-image-btn');

        // Initialize a counter to track which image to show next

        let currentImageIndex = 0;


        // Add a 'click' event listener to the button

        showImageBtn.addEventListener('click', () => {

            // Check if there are still images to show

            if (currentImageIndex < images.length) {

                // Show the next image by removing the 'hidden' class

                images[currentImageIndex].classList.remove('hidden');

                // Increment the counter

                currentImageIndex++;

            }


            // If all images have been shown, hide the button

            if (currentImageIndex >= images.length) {

                showImageBtn.classList.add('hidden');

            }

        });

    </script>
